int main() {
	/********* Begin **********/
	/*用while循环实现百内求和*/
	int i = 1, sum = 0;  // 错误1：这里用的是中文逗号，应该用英文逗号
	while (i<=100) {     // 错误2：这里用的是中文括号），应该用英文括号)
	    sum += i;
	    i++;
	}
	cout << "sum = " << sum << endl;
	
	/*用do-while循环实现内求和*/
	i = 1;
	sum = 0;
	do {
	    sum += i;        // 错误3：这里用的是中文分号；，应该用英文分号;
	    i++;
	} while (i <= 100);
	cout << "sum = " << sum << endl;

}

	while (i <= 100) {
	    sum += i;
	    i += 2;  // 每次增加2，保持为奇数
	}
	cout << "sum = " << sum << endl;
	
	/********** End **********/

#include <iostream>
using namespace std;
/*注意代码只可以在"Begin……End"输入，不得修改除"Begin……End"外的任何代码。*/

int main() {
	int n = 0, score, max = 0;    //人数，得分，最高分
	float avg = 0;            //平均分
	/************* Begin ************/
	int sum = 0;  // 总分
	
	while (true) {
		n++;  // 输入次数计数
		cout << "请输入第" << n << "个得分[0,100]：";
		cin >> score;
		
		if (score == -1) {
			n--;  // 减去-1这次输入
			break;  // 结束输入
		}
		
		sum += score;  // 累加总分
		
		if (score > max) {
			max = score;  // 更新最高分
		}
	}
	
	if (n > 0) {
		avg = (float)sum / n;  // 计算平均分
	}
	
	/************* End ************/
	cout << "最高分：" << max << "\n平均分：" << avg << endl;
	return 0;
}
#include <iostream>
using namespace std;

int main() {
	/********* Begin **********/
	int sum = 0;  // 总和
	
	// 遍历奇数：1, 3, 5, 7, 9
	for (int i = 1; i <= 9; i += 2) {
		int factorial = 1;  // 阶乘结果
		
		// 计算i的阶乘
		for (int j = 1; j <= i; j++) {
			factorial *= j;
		}
		
		sum += factorial;  // 累加到总和
	}
	
	cout << "sum = " << sum << endl;
	
	/********** End **********/
	return 0;
}
#include <iostream>
using namespace std;

int main() {
	/********* Begin **********/
	int count = 0;  // 水仙花数的个数
	
	// 遍历所有3位数
	for (int i = 100; i < 1000; i++) {
		// 分离出百位、十位、个位
		int bai = i / 100;        // 百位
		int shi = (i / 10) % 10;  // 十位
		int ge = i % 10;          // 个位
		
		// 判断是否为水仙花数
		if (bai*bai*bai + shi*shi*shi + ge*ge*ge == i) {
			cout << i << endl;
			count++;
		}
	}
	
	cout << "100-1000的水仙花数有：" << count << "个" << endl;
	
	/********** End **********/
	return 0;
}
#include <iostream>

int main() {
    int den = 1, sign = 1;
    double item = 1.0, pi = 0.0;
    while (fabs(item) >= 1e-8) {
        pi += item;
        den += 2;
        sign = -sign;
        item = sign / double(den);
    }
    cout << "pi = " << fixed << setprecision(7) << (pi * 4) << endl;
    return 0;
}
#include <iostream>
#include <iomanip>

int main()
{
    int cow[21];          // 存放前20项
    cow[1] = cow[2] = cow[3] = 1;

    for (int i = 4; i <= 20; i++)
        cow[i] = cow[i - 1] + cow[i - 3];  // 递推公式

    for (int i = 1; i <= 20; i++) {
        cout << setw(6) << cow[i];
        if (i % 4 == 0) cout << endl;      // 每4个换行
    }

    return 0;
}
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;
    cout << "请输入>1的正整数n：";
    cin >> n;

    int count = 0; // 用来统计当前已输出的素数个数

    for (int i = 2; i <= n; i++) {  // 遍历 2~n
        bool isPrime = true;        // 假设 i 是素数
        for (int j = 2; j * j <= i; j++) {  // 判断是否有因子
            if (i % j == 0) {
                isPrime = false;
                break;
            }
        }

        if (isPrime) {
            cout << setw(5) << i;   // 右对齐输出
            count++;
            if (count % 5 == 0)     // 每 5 个换一行
                cout << endl;
        }
    }

    cout << endl;
    return 0;
}
#include <iostream>

int main()
{
    int i, j;

    // 打印表头
    cout << "    |";
    for (j = 1; j <= 9; j++)
        cout << setw(4) << j;
    cout << "\n----|------------------------------------\n";

    // 打印乘法表
    for (i = 1; i <= 9; i++) {
        cout << i << "   |"; 
        for (j = 1; j <= 9; j++) {
            if (j < i)
                cout << setw(4) << " ";
            else
                cout << setw(4) << i * j;
        }
        cout << endl;
    }

    return 0;
}
